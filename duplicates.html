Prevent duplicate assets

Stop duplicate asset and model records, while providing a nice UX.

March 28, 2019 JGriffin

In order to prevent duplicate records in ServiceNow, you can simply enable the Unique Dictionary value for the table column that you wish to be unique. The only "problem" with only doing this approach is is the user expereince isn't ideal since you will be presented with a "rude" error message. The below Business Rule scripts were created in addition to setting the alm_hardware, serial_number column as Unique.

ALM_Hardware Script Condition: gs.getSession().isInteractive() && current.serial_number.changes();.
CMDB_Hardware_Product_Model_List Script Condition: gs.getSession().isInteractive() && current.model_number.changes() && current.name.changes();

Business Rule:

									
(function executeRule(current, previous /*null when async*/) {

	var grAsset = new GlideRecord('alm_hardware');
	grAsset.addQuery('serial_number', current.serial_number);
	grAsset.query();

	if (grAsset.next() && grAsset.sys_id != current.sys_id) { 
		gs.addErrorMessage('Error Saving: Asset with matching serial number already exists!');
		current.setAbortAction(true);
	}

})(current, previous);

Hardware Model - Business Rule (before):

									
(function executeRule(current, previous /*null when async*/) {

	var grModel = new GlideRecord('cmdb_hardware_product_model');
	if (!current.model_number) {
		grModel.addQuery('name', current.name);
		grModel.query();
	} else {
		grModel.addQuery('name', current.name).addOrCondition('model_number', current.model_number);
		grModel.addOrCondition('model_number', current.model_number);
		grModel.query();
	}

	if (grModel.next() && grModel.sys_id != current.sys_id) { 
		gs.addErrorMessage('Error Saving: Hardware model with matching name/number already exists!');
		current.setAbortAction(true);
	}

})(current, previous);
